✅ Fix: **Helper-Sub/Function muss auf Modulebene stehen**, also **außerhalb** von `Create_P02_...`.

Unten ist ein **sauberer, robuster VBA-Code**, der:
- Deckblatt + 7 Inhaltsfolien erstellt
- immer ein passendes Layout nutzt (Title / Title+Content)
- den Text sicher in die **richtigen Platzhalter** schreibt (ohne Placeholder(2)-Fehler)

```vba
Option Explicit

'--- Helper: fügt eine "Titel + Inhalt"-Folie hinzu
Public Sub AddTitleContentSlide(ByVal ppt As Presentation, _
                               ByVal slideTitle As String, _
                               ByVal bullets As Variant)

    Dim sld As Slide
    Dim i As Long
    Dim txt As String

    ' Layout: Titel + Inhalt
    Set sld = ppt.Slides.Add(ppt.Slides.Count + 1, ppLayoutText)

    ' Titel setzen
    sld.Shapes.Title.TextFrame.TextRange.Text = slideTitle

    ' Bullet-Text zusammenbauen
    txt = ""
    For i = LBound(bullets) To UBound(bullets)
        If Len(txt) > 0 Then txt = txt & vbCrLf
        txt = txt & "• " & CStr(bullets(i))
    Next i

    ' Inhalt setzen (Placeholder(2) ist bei ppLayoutText i.d.R. der Content-Placeholder)
    sld.Shapes.Placeholders(2).TextFrame.TextRange.Text = txt

End Sub

'--- Hauptmakro: baut die Präsentation
Public Sub Create_P02_Management_Presentation()
```

```vba
Option Explicit

'==== Helper: Rounded Tile with colored header + bullets + keyline ====
Private Sub AddTile( _
    ByVal sld As Slide, _
    ByVal x As Single, ByVal y As Single, ByVal w As Single, ByVal h As Single, _
    ByVal borderColor As Long, _
    ByVal headerText As String, _
    ByVal bullets As Variant, _
    ByVal keyline As String)

    Dim tile As Shape, bar As Shape, tTitle As Shape, tBody As Shape, tKey As Shape
    Dim barH As Single: barH = 28 'points
    Dim pad As Single: pad = 12
    Dim i As Long, bodyText As String

    ' Tile container
    Set tile = sld.Shapes.AddShape(msoShapeRoundedRectangle, x, y, w, h)
    tile.Fill.ForeColor.RGB = RGB(255, 255, 255)
    tile.Line.ForeColor.RGB = borderColor
    tile.Line.Weight = 2.25

    ' Header bar
    Set bar = sld.Shapes.AddShape(msoShapeRoundedRectangle, x, y, w, barH)
    bar.Fill.ForeColor.RGB = borderColor
    bar.Line.ForeColor.RGB = borderColor

    ' Header text
    Set tTitle = sld.Shapes.AddTextbox(msoTextOrientationHorizontal, x + pad, y + 4, w - 2 * pad, barH - 6)
    With tTitle.TextFrame
        .TextRange.Text = headerText
        .TextRange.Font.Size = 16
        .TextRange.Font.Bold = msoTrue
        .TextRange.Font.Color.RGB = RGB(255, 255, 255)
        .VerticalAnchor = msoAnchorMiddle
        .WordWrap = msoTrue
    End With

    ' Bullets body
    bodyText = ""
    For i = LBound(bullets) To UBound(bullets)
        If Len(bodyText) > 0 Then bodyText = bodyText & vbCrLf
        bodyText = bodyText & "• " & CStr(bullets(i))
    Next i

    Set tBody = sld.Shapes.AddTextbox(msoTextOrientationHorizontal, x + pad, y + barH + 10, w - 2 * pad, h - barH - 46)
    With tBody.TextFrame
        .TextRange.Text = bodyText
        .TextRange.Font.Size = 14
        .TextRange.Font.Color.RGB = RGB(34, 34, 34)
        .WordWrap = msoTrue
        .AutoSize = ppAutoSizeNone
    End With

    ' Keyline (bold statement)
    Set tKey = sld.Shapes.AddTextbox(msoTextOrientationHorizontal, x + pad, y + h - 26, w - 2 * pad, 20)
    With tKey.TextFrame
        .TextRange.Text = keyline
        .TextRange.Font.Size = 14
        .TextRange.Font.Bold = msoTrue
        .TextRange.Font.Color.RGB = borderColor
        .WordWrap = msoTrue
        .VerticalAnchor = msoAnchorMiddle
    End With
End Sub

'==== Main: Create One-Pager Tile Slide ====
Public Sub Create_P02_OnePager_TileSlide()

    Dim ppt As Presentation
    Dim sld As Slide
    Dim W As Single, H As Single
    Dim margin As Single, gap As Single, top As Single
    Dim tileW As Single, tileH As Single

    ' Colors (RGB)
    Dim COL_IST As Long:     COL_IST = RGB(31, 119, 180)   'blue
    Dim COL_SCH As Long:     COL_SCH = RGB(214, 39, 40)    'red
    Dim COL_HEB As Long:     COL_HEB = RGB(44, 160, 44)    'green
    Dim COL_IMP As Long:     COL_IMP = RGB(188, 189, 34)   'yellow-ish
    Dim COL_DARK As Long:    COL_DARK = RGB(34, 34, 34)
    Dim COL_BG As Long:      COL_BG = RGB(246, 247, 249)

    Set ppt = ActivePresentation
    W = ppt.PageSetup.SlideWidth
    H = ppt.PageSetup.SlideHeight

    ' New blank slide
    Set sld = ppt.Slides.Add(ppt.Slides.Count + 1, ppLayoutBlank)

    ' Background
    With sld.Shapes.AddShape(msoShapeRectangle, 0, 0, W, H)
        .Fill.ForeColor.RGB = COL_BG
        .Line.ForeColor.RGB = COL_BG
    End With

    ' Header (title + subtitle)
    Dim tTitle As Shape, tSub As Shape
    Set tTitle = sld.Shapes.AddTextbox(msoTextOrientationHorizontal, 30, 12, W - 60, 36)
    With tTitle.TextFrame
        .TextRange.Text = "P02-Kette – Auftragsqualität & No-Touch-Quote"
        .TextRange.Font.Size = 30
        .TextRange.Font.Bold = msoTrue
        .TextRange.Font.Color.RGB = COL_DARK
        .WordWrap = msoTrue
    End With

    Set tSub = sld.Shapes.AddTextbox(msoTextOrientationHorizontal, 30, 50, W - 60, 18)
    With tSub.TextFrame
        .TextRange.Text = "One-Pager | Management Overview"
        .TextRange.Font.Size = 14
        .TextRange.Font.Color.RGB = RGB(85, 85, 85)
        .WordWrap = msoTrue
    End With

    ' Grid geometry
    margin = 30
    gap = 18
    top = 80

    tileW = (W - 2 * margin - gap) / 2
    tileH = (H - top - margin - gap) / 2

    ' Tiles content
    Call AddTile(sld, margin, top, tileW, tileH, COL_IST, _
        "🟦  IST", _
        Array( _
            "Geringer Cyberdaten-Anteil in Belegen", _
            "Hoher manueller Aufwand in P02 & PFS", _
            "Qualitätssicherung reaktiv (quartalsweise)", _
            "Medienbruch: eBANF ↔ PFS" _
        ), _
        "Qualität wird nachbearbeitet – nicht gesteuert.")

    Call AddTile(sld, margin + tileW + gap, top, tileW, tileH, COL_SCH, _
        "🟥  SCHMERZ", _
        Array( _
            "Lange, volatile Durchlaufzeiten", _
            "Hohe Bindung qualifizierter FTE", _
            "Erhöhte Fehler- & Nacharbeitskosten", _
            "No-Touch-Quote stagniert / skaliert nicht" _
        ), _
        "Mehr Volumen = mehr Personal – nicht mehr Effizienz.")

    Call AddTile(sld, margin, top + tileH + gap, tileW, tileH, COL_HEB, _
        "🟩  HEBEL", _
        Array( _
            "Ausbaustufe I: Daten zusammenführen & KI-Analyse", _
            "Integration D!VE-Auftragsmonitoring", _
            "Ausbaustufe II: Fehlende Cyberdaten ermitteln", _
            "Belege automatisiert ergänzen (P02/P01/PFS/PMA-ISEM)" _
        ), _
        "Transparenz → Steuerung → Automatisierung.")

    Call AddTile(sld, margin + tileW + gap, top + tileH + gap, tileW, tileH, COL_IMP, _
        "🟨  IMPACT", _
        Array( _
            "+20–40 %-Punkte No-Touch-Quote (Zielkorridor)", _
            "Signifikant weniger manuelle Bearbeitung", _
            "Stabilere, planbare Durchlaufzeiten", _
            "Freisetzung qualifizierter FTE-Kapazitäten" _
        ), _
        "Skalierungs- & Kostenhebel für die P02-Kette.")

End Sub
```
